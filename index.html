import os import time import pickle import moviepy.editor as mp from gtts import gTTS import requests from google.oauth2.credentials import Credentials from google_auth_oauthlib.flow import InstalledAppFlow from google.auth.transport.requests import Request from googleapiclient.discovery import build from googleapiclient.http import MediaFileUpload # --- CONFIGURATION --- SCOPES = ['https://www.googleapis.com/auth/youtube.upload'] VIDEO_TITLE = "Amazing Facts about Python #shorts" VIDEO_DESCRIPTION = "This video was created automatically using AI and Python." CATEGORY_ID = "22" # People & Blogs KEYWORDS = ["python", "ai", "automation", "facts"] VISIBILITY = "public" # public or private # 1. GENERATE AI SCRIPT (Simulated) def generate_script(topic): # In a real app, you would call OpenAI API here. # For now, we use a static script. script = f"Here are 5 amazing facts about {topic}. Fact one: Python is named after Monty Python, not the snake." return script # 2. CREATE VIDEO (Audio + Visuals) def create_video(text_script, output_filename): print("Step 1: Creating Audio...") # Convert Text to Speech (gTTS) tts = gTTS(text=text_script, lang='en') audio_filename = "voiceover.mp3" tts.save(audio_filename) print("Step 2: Downloading Background Image...") # Download a random tech image from Unsplash (Free source) # You can replace this URL with any image URL image_url = "https://images.unsplash.com/photo-1526374965328-7f61d4dc18c5?w=800&q=80" img_data = requests.get(image_url).content with open('background.jpg', 'wb') as handler: handler.write(img_data) print("Step 3: Assembling Video with MoviePy...") # Load Image clip = mp.ImageClip("background.jpg") # Set duration to match audio length audio_clip = mp.AudioFileClip(audio_filename) clip = clip.set_duration(audio_clip.duration) # Add Audio final_clip = clip.set_audio(audio_clip) # Write video file final_clip.write_videofile(output_filename, fps=24) print(f"Video created: {output_filename}") # 3. AUTHENTICATE & UPLOAD def upload_to_youtube(video_file, title, description): print("Step 4: Authenticating with YouTube...") creds = None # The file token.pickle stores the user's access and refresh tokens if os.path.exists('token.pickle'): with open('token.pickle', 'rb') as token: creds = pickle.load(token) # If there are no (valid) credentials available, let the user log in. if not creds or not creds.valid: if creds and creds.expired and creds.refresh_token: creds.refresh(Request()) else: flow = InstalledAppFlow.from_client_secrets_file( 'client_secrets.json', SCOPES) creds = flow.run_local_server(port=0) # Save the credentials for the next run with open('token.pickle', 'wb') as token: pickle.dump(creds, token) youtube = build('youtube', 'v3', credentials=creds) print("Step 5: Uploading Video...") request = youtube.videos().insert( part="snippet,status", body={ "snippet": { "title": title, "description": description, "categoryId": CATEGORY_ID, "tags": KEYWORDS }, "status": { "privacyStatus": VISIBILITY } }, media_body=MediaFileUpload(video_file) ) response = request.execute() print(f"Upload Successful! Video ID: {response['id']}") # --- MAIN EXECUTION --- if __name__ == "__main__": topic = "Python Programming" output_video = "ai_video.mp4" # Run the pipeline script = generate_script(topic) create_video(script, output_video) upload_to_youtube(output_video, VIDEO_TITLE, VIDEO_DESCRIPTION)
